üîß Vue Plugin Setup

1. Install Plugin

// main.js import { createApp } from 'vue' import App from './App.vue' import Notifio from 'notifio/vue' const app = createApp(App) app.use(Notifio) app.mount('#app')

2. Use in Components

// In your Vue components <template> <div> <button @click="showSuccess">Show Success</button> <button @click="showError">Show Error</button> </div> </template> <script setup> import { useNotifio } from 'notifio/vue' const notifio = useNotifio() const showSuccess = () => { notifio.success('Operation completed!') } const showError = () => { notifio.error('Something went wrong!') } </script>

üéØ Vue Composition API

useNotifio Composable

The useNotifio composable provides reactive access to all notification methods.

Reactive Notifications

// Reactive notification example <script setup> import { ref, computed } from 'vue' import { useNotifio } from 'notifio/vue' const notifio = useNotifio() const count = ref(0) const showReactiveNotification = () => { count.value++ notifio.info(`Count is now: ${count.value}`) } // Computed notification const message = computed(() => `Current time: ${new Date().toLocaleTimeString()}`) const showComputedNotification = () => { notifio.success(message.value) } </script>

Plugin Installation

Composable Usage

// Import the composable import { useNotifio } from 'notifio/vue' // Use in setup function const notifio = useNotifio() // All methods are available notifio.success('Success!') notifio.error('Error!') notifio.warning('Warning!') notifio.info('Info!') notifio.neutral('Neutral!')

TypeScript Support

// TypeScript usage import { useNotifio, type NotificationOptions } from 'notifio/vue' const notifio = useNotifio() const options: NotificationOptions = { title: 'Success', description: 'Operation completed', duration: 5000, position: 'top-right' } notifio.success('Message', options)
